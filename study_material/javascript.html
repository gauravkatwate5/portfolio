Core Concepts

Variables: var, let, const

var: Function-scoped, hoisted, can be redeclared. let: Block-scoped, not hoisted, temporal dead zone. const: Block-scoped, cannot be reassigned (but object properties can).

var a = 1;    // Hoisted, redeclarable
let b = 2;    // Block scope, TDZ
const c = 3;  // Cannot reassign

const obj = { x: 10 };
obj.x = 20;   // Allowed
// obj = {}   // Error

Data Types & Operators

Primitive: String, Number, Boolean, Null, Undefined, Symbol, BigInt. Non-primitive: Object.

typeof null;        // "object" (historical bug)
typeof [];          // "object"
typeof function(){};// "function"

1 == '1'   // true (coercion)
1 === '1'  // false (strict)

Functions, Scope, Hoisting

Functions are hoisted with definition. Variables with var are hoisted with undefined.

console.log(hoist); // undefined
var hoist = 'Hi';

sayHello(); // Works
function sayHello() { console.log('Hello'); }

// Arrow functions not hoisted
// greet(); // ReferenceError
const greet = () => console.log('Hi');

Closures

A function that remembers its outer variables even after the outer function has finished executing.

function outer() {
  let count = 0;
  return function inner() {
    count++;
    return count;
  };
}
const counter = outer();
console.log(counter()); // 1
console.log(counter()); // 2

IIFE (Immediately Invoked Function Expression)

Creates private scope, avoids polluting global namespace.

})();
// private is not accessible here

Debouncing

Limit function calls after user stops typing.

function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

Throttling

Limit function calls to once every X ms.

function throttle(fn, limit) {
  let inThrottle;
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

Coding Challenges (with Solutions)

1. Reverse a String

Return the string in reverse order without using built-in reverse.

function reverseString(str) {
  let reversed = '';
  for (let i = str.length - 1; i >= 0; i--) {
    reversed += str[i];
  }
  return reversed;
}
// Or: [...str].reverse().join('')

2. Check Palindrome

Return true if string reads same forward and backward (ignore case, spaces).

function isPalindrome(str) {
  const clean = str.toLowerCase().replace(/[^a-z0-9]/g, '');
  return clean === clean.split('').reverse().join('');
}

3. Find Duplicates in Array

Return array of elements that appear more than once.

function findDuplicates(arr) {
  const seen = new Set();
  const duplicates = new Set();
  arr.forEach(item => {
    if (seen.has(item)) duplicates.add(item);
    else seen.add(item);
  });
  return [...duplicates];
}

4. Flatten Nested Array

Flatten any level of nesting.

function flatten(arr) {
  return arr.flat(Infinity);
}
// Or recursive:
function flattenDeep(arr) {
  return arr.reduce((acc, val) => 
    Array.isArray(val) ? acc.concat(flattenDeep(val)) : acc.concat(val), []);
}

5. Implement Debounce

Delay function call until after wait milliseconds have elapsed.

function debounce(func, wait) {
  let timeout;
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
}

6. Get Unique Values

Remove duplicates from array.

const unique = [...new Set(arr)];
// Or: Array.from(new Set(arr))

7. Sum of Digits

Return sum of all digits in a number.

function sumDigits(n) {
  return Math.abs(n).toString().split('').reduce((a,b) => a + +b, 0);
}

8. Anagram Check

Check if two strings are anagrams.

function areAnagrams(s1, s2) {
  const normalize = s => s.toLowerCase().replace(/[^a-z0-9]/g, '').split('').sort().join('');
  return normalize(s1) === normalize(s2);
}

9. Currying

Convert f(a,b,c) → f(a)(b)(c)

function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn(...args);
    }
    return (...next) => curried(...args, ...next);
  };
}

10. Deep Clone

Clone nested objects.

function deepClone(obj) {
  if (obj === null || typeof obj !== 'object') return obj;
  if (obj instanceof Date) return new Date(obj);
  if (obj instanceof Array) return obj.map(deepClone);
  if (typeof obj === 'object') {
    const cloned = {};
    for (let key in obj) {
      if (obj.hasOwnProperty(key)) {
        cloned[key] = deepClone(obj[key]);
      }
    }
    return cloned;
  }
}

11. Fibonacci (Memoized)

const fib = (n, memo = {}) => {
  if (n in memo) return memo[n];
  if (n <= 1) return n;
  memo[n] = fib(n-1, memo) + fib(n-2, memo);
  return memo[n];
};

12. Group By

function groupBy(arr, key) {
  return arr.reduce((acc, obj) => {
    const k = obj[key];
    acc[k] = acc[k] || [];
    acc[k].push(obj);
    return acc;
  }, {});
}

13. Promise.all Implementation

function promiseAll(promises) {
  return new Promise((resolve, reject) => {
    let results = [];
    let completed = 0;
    if (promises.length === 0) resolve([]);
    promises.forEach((p, i) => {
      Promise.resolve(p).then(val => {
        results[i] = val;
        completed++;
        if (completed === promises.length) resolve(results);
      }).catch(reject);
    });
  });
}

14. Memoize Function

function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn(...args);
    cache.set(key, result);
    return result;
  };
}

15. Event Emitter

class EventEmitter {
  constructor() { this.events = {}; }
  on(event, callback) {
    this.events[event] = this.events[event] || [];
    this.events[event].push(callback);
  }
  emit(event, ...args) {
    if (this.events[event]) {
      this.events[event].forEach(cb => cb(...args));
    }
  }
}

60 Full Interview Questions with Code & Explanations

Q1. What is Hoisting in JavaScript?

Hoisting is JavaScript’s default behavior of moving variable and function declarations to the top of their containing scope during the compilation phase. Function declarations are fully hoisted, var variables are hoisted and initialized with undefined, while let and const are hoisted but not initialized (Temporal Dead Zone).

console.log(x); // undefined
var x = 5;

console.log(y); // ReferenceError: Cannot access 'y' before initialization
let y = 10;

greet(); // "Hello"
function greet() { console.log('Hello'); }

Q2. Difference between == and ===?

== performs type coercion (converts types before comparison). === is strict equality — checks both value and type without coercion.

1 == '1'     // true (string converted to number)
1 === '1'    // false

null == undefined   // true
null === undefined  // false

0 == false   // true
0 === false  // false

Q3. What is a Closure? Give a real-world use case.

A closure is a function that retains access to its lexical scope even when the function is executed outside that scope. Used for data privacy, module pattern, and event handlers.

function createCounter() {
  let count = 0;
  return {
    increment: () => ++count,
    getCount: () => count
  };
}
const counter = createCounter();
counter.increment(); // 1
counter.increment(); // 2
console.log(counter.getCount()); // 2

Q4. Explain this keyword in different contexts.

this refers to the object executing the current function:

  • Global context: window (browser)
  • Object method: the object
  • Constructor: new instance
  • Event handler: the DOM element
  • Arrow function: inherits from parent scope
const person = {
  name: 'Gaurav',
  greet: function() { console.log(this.name); }
};
person.greet(); // "Gaurav"

const fn = person.greet;
fn(); // undefined (lost context)

Q5. What is the Event Loop?

The event loop continuously checks the call stack. If empty, it takes the first callback from the task queue (macrotasks) or microtask queue (Promises) and pushes it to the stack. Microtasks run before macrotasks.

console.log('1');
setTimeout(() => console.log('2'), 0);
Promise.resolve().then(() => console.log('3'));
console.log('4');
// Output: 1 → 4 → 3 → 2

Q6. What are Promises? Explain states.

Promises represent the eventual result of an asynchronous operation. States: pending, fulfilled (with value), rejected (with reason).

const p = new Promise((resolve, reject) => {
  setTimeout(() => resolve('Success'), 1000);
});
p.then(result => console.log(result))
 .catch(err => console.error(err));

Q7. What is async/await?

Syntactic sugar over Promises that makes asynchronous code look and behave like synchronous code.

async function fetchUser() {
  try {
    const res = await fetch('/api/user');
    const user = await res.json();
    return user;
  } catch (err) {
    console.error('Error:', err);
  }
}

Q8. What is Prototypal Inheritance?

Objects inherit properties and methods from a prototype object via the [[Prototype]] chain. Every object has a __proto__ link to its constructor’s prototype.

function Animal(name) { this.name = name; }
Animal.prototype.speak = function() { return 'sound'; };
const dog = new Animal('Dog');
console.log(dog.speak()); // "sound"

Q9. Explain call, apply, bind.

All set the this value explicitly:

  • call(thisArg, a, b, c): invokes immediately with args
  • apply(thisArg, [a, b, c]): invokes with array of args
  • bind(thisArg): returns new function with bound this
function greet(greeting) {
  console.log(\`\${greeting}, \${this.name}\`);
}
const person = { name: 'Gaurav' };
greet.call(person, 'Hello');
greet.apply(person, ['Hi']);
const bound = greet.bind(person);
bound('Hey');

Q10. What is Event Bubbling?

When an event occurs on a DOM element, it first triggers on the target, then bubbles up through ancestors. Use stopPropagation() to prevent.

child.addEventListener('click', () => console.log('Child'));
parent.addEventListener('click', () => console.log('Parent'));
// Click child → logs: Child → Parent

Q11. What is debounce vs throttle?

Debounce: Waits for a pause in events before executing (e.g., search input). Throttle: Executes at most once every X ms (e.g., scroll).

// Debounce: wait 300ms after last keypress
// Throttle: fire every 200ms during scroll

Q12. What is Optional Chaining?

Allows safe access to nested object properties. Returns undefined instead of throwing error if intermediate property is null/undefined.

const name = user?.profile?.name; // undefined if user or profile is null

Q13. What is Nullish Coalescing?

Returns right operand only if left is null or undefined. Unlike || which treats all falsy values.

const input = value ?? 'default'; // Only if null/undefined
const input2 = value || 'default'; // Also if 0, '', false

Q14. How to deep clone an object?

Use recursive function or structuredClone() (modern). JSON.parse(JSON.stringify()) fails with functions, dates, circular refs.

function deepClone(obj) {
  if (obj === null || typeof obj !== 'object') return obj;
  const clone = Array.isArray(obj) ? [] : {};
  for (let key in obj) {
    clone[key] = deepClone(obj[key]);
  }
  return clone;
}

Q15. What is Symbol?

Primitive type for unique, immutable identifiers. Used as object keys to avoid name collisions.

Q16. What is typeof null?

Returns "object" — a historical bug in JavaScript that cannot be fixed due to backward compatibility.

console.log(typeof null); // "object"

Q17. What is the difference between undefined and not defined?

undefined: variable declared but not assigned. not defined: variable never declared (ReferenceError).

let x;
console.log(x); // undefined

console.log(y); // ReferenceError: y is not defined

Q18. What is NaN? How to check?

Not-a-Number. Result of invalid math. Use isNaN() or Number.isNaN().

0 / 0; // NaN
Number.isNaN(NaN); // true
isNaN('hello'); // true (coerces)

Q19. What is the difference between null and undefined?

null: intentional absence of value (object). undefined: uninitialized variable or missing property.

let a = null;      // explicit
let b;             // implicit undefined
obj.missingProp;   // undefined

Q20. What are IIFEs? Why use them?

Immediately Invoked Function Expressions create private scope and avoid polluting global namespace.

(function() {
  var privateVar = 'secret';
  console.log(privateVar);
})();
// privateVar is not accessible

Q21. What is the Temporal Dead Zone (TDZ)?

Time between entering scope and initialization of let/const variables. Accessing them throws ReferenceError.

console.log(x); // ReferenceError
let x = 10;

Q22. What is strict mode?

Enforces stricter parsing and error handling. Eliminates silent errors, improves security.

'use strict';
x = 10; // ReferenceError (no var/let/const)

Q23. What is currying?

Transforming a function with multiple arguments into a sequence of functions that take one argument each.

function add(a) {
  return function(b) {
    return a + b;
  };
}
add(2)(3); // 5

Q24. What is memoization?

Caching function results based on input arguments to avoid recomputation.

function memoize(fn) {
  const cache = {};
  return function(...args) {
    const key = JSON.stringify(args);
    if (key in cache) return cache[key];
    return cache[key] = fn(...args);
  };
}

Q25. What is the difference between for...in and for...of?

for...in: iterates over enumerable property names (objects). for...of: iterates over iterable values (arrays, strings).

const obj = {a:1, b:2};
for (let key in obj) console.log(key); // "a", "b"

const arr = [1,2,3];
for (let val of arr) console.log(val); // 1,2,3

Q26. What is the spread operator?

Expands iterables into individual elements. Used in arrays, objects, function args.

const arr = [1,2,3];
console.log(...arr); // 1 2 3

const obj = {a:1};
const clone = {...obj};

Q27. What is the rest parameter?

Collects remaining arguments into an array. Must be last parameter.

function sum(...numbers) {
  return numbers.reduce((a,b) => a+b);
}
sum(1,2,3,4); // 10

Q28. What is destructuring?

Extract values from arrays/objects into distinct variables.

const [a, b] = [1, 2];
const {name, age} = user;

Q29. What is Promise.all()?

Runs multiple promises in parallel. Resolves with array of results when all complete. Rejects if any fails.

Promise.all([p1, p2, p3])
  .then(results => console.log(results))
  .catch(err => console.error(err));

Q30. What is Promise.race()?

Resolves or rejects as soon as one promise settles.

Promise.race([slow, fast]).then(winner => console.log(winner));

Q31. What is async function return value?

Always returns a Promise. If you return a value, it’s wrapped in resolved Promise.

async function f() { return 1; }
f().then(v => console.log(v)); // 1

Q32. How to handle errors in async/await?

Q39. What is JSON.stringify() and JSON.parse()?

Q40. What is a memory leak in JS? How to avoid?

Memory not released when no longer needed. Causes: global vars, forgotten event listeners, setInterval without clearInterval.

// Bad

// Good

Q41. What is Object.freeze()?

Makes an object immutable — prevents adding, removing, or modifying properties.

Prevents adding/removing properties but allows modifying existing ones.
